<script>
// бҹЎ. бһҳбһ»бһҒбһ„бһ¶бһҡбһҹбҹ’бһҖбҹҒбһ“бһҡбһјбһ”бһ—бһ¶бһ– (OCR)
async function scanImage() {
    const fileInput = document.getElementById('imageInput');
    const ocrStatus = document.getElementById('ocrStatus');
    const chineseText = document.getElementById('chineseText');

    if (fileInput.files.length === 0) return;

    ocrStatus.innerText = "вҸі бһҖбҹҶбһ–бһ»бһ„бһҹбҹ’бһҖбҹҒбһ“бһҡбһјбһ”бһ—бһ¶бһ– бһ“бһ·бһ„бһҹбҹ’бһҡбһ„бҹӢбһўбһҖбҹ’бһҹбһҡбһ…бһ·бһ“... бһҹбһјбһҳбһҡбһ„бҹӢбһ…бһ¶бҹҶбһҳбһҪбһҷбһ—бҹ’бһӣбҹӮбһҸбҹ”";
    
    try {
        // бһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢ Tesseract.js бһҹбҹ’бһҖбҹҒбһ“бһҡбһҖбһ—бһ¶бһҹбһ¶бһ…бһ·бһ“бһҹбһ¶бһҳбһүбҹ’бһү (chi_sim)
        const result = await Tesseract.recognize(fileInput.files[0], 'chi_sim');
        chineseText.value = result.data.text;
        ocrStatus.innerText = "вң… бһҹбҹ’бһҖбҹҒбһ“бһҮбҹ„бһӮбһҮбҹҗбһҷ!";
    } catch (error) {
        ocrStatus.innerText = "вқҢ бһҳбһ¶бһ“бһ”бһүбҹ’бһ бһ¶бһҖбҹ’бһ“бһ»бһ„бһҖбһ¶бһҡбһҹбҹ’бһҖбҹҒбһ“бһҡбһјбһ”бһ—бһ¶бһ–бҹ”";
        console.error(error);
    }
}

// бҹў. бһҳбһ»бһҒбһ„бһ¶бһҡбһўбһ¶бһ“бһҹбҹҶбһЎбҹҒбһ„бһ—бһ¶бһҹбһ¶бһ…бһ·бһ“ (Speech)
function speakChinese() {
    let text = document.getElementById("chineseText").value;
    if(!text.trim()) { alert("бһҹбһјбһҳбһ”бһүбҹ’бһ…бһјбһӣбһўбһҖбҹ’бһҹбһҡбһ…бһ·бһ“бһҮбһ¶бһҳбһ»бһ“бһҹбһ·бһ“!"); return; }

    let utterance = new SpeechSynthesisUtterance(text);
    utterance.lang = 'zh-CN'; 
    utterance.rate = 0.75; // бһӣбҹ’бһ”бһҝбһ“бһўбһ¶бһ“бһҷбһәбһҸбһӣбҹ’бһҳбһҳбһҹбҹ’бһҸбһ¶бһ”бҹӢбһ‘бһ¶бһ“бҹӢ
    window.speechSynthesis.speak(utterance);
}

// бҹЈ. бһҳбһ»бһҒбһ„бһ¶бһҡбһ”бҹҶбһ”бҹ’бһӣбҹӮбһ„бһҮбһ¶бһ—бһёбһ“бһўбҹҠбһ·бһ“ (Pinyin)
function generatePinyin() {
    let text = document.getElementById("chineseText").value;
    if(!text.trim()) { alert("бһҹбһјбһҳбһ”бһүбҹ’бһ…бһјбһӣбһўбһҖбҹ’бһҹбһҡбһ…бһ·бһ“бһҮбһ¶бһҳбһ»бһ“бһҹбһ·бһ“!"); return; }

    document.getElementById("resultSection").style.display = "block";
    // бһ”бҹ’бһҡбһҫбһ”бһҺбҹ’бһҺбһ¶бһӣбҹҗбһҷ pinyinPro бһҠбһҫбһҳбҹ’бһ”бһёбһ”бҹҶбһ”бҹ’бһӣбҹӮбһ„
    const pinyinResult = pinyinPro.pinyin(text, { toneType: 'symbol' }); 
    document.getElementById("pinyinResult").innerText = pinyinResult;
}

// бҹӨ. бһҳбһ»бһҒбһ„бһ¶бһҡбһ”бһҖбһ”бҹ’бһҡбҹӮбһҮбһ¶бһ—бһ¶бһҹбһ¶бһҒбҹ’бһҳбҹӮбһҡ (Translation)
async function translateToKhmer() {
    let text = document.getElementById("chineseText").value;
    if(!text.trim()) { alert("бһҹбһјбһҳбһ”бһүбҹ’бһ…бһјбһӣбһўбһҖбҹ’бһҹбһҡбһ…бһ·бһ“бһҮбһ¶бһҳбһ»бһ“бһҹбһ·бһ“!"); return; }

    document.getElementById("resultSection").style.display = "block";
    document.getElementById("translateResult").innerText = "вҸі бһҖбҹҶбһ–бһ»бһ„бһ”бһҖбһ”бҹ’бһҡбҹӮ...";

    try {
        // бһ”бҹ’бһҡбһҫбһ”бҹ’бһҡбһ¶бһҹбҹӢ API бһ”бһҖбһ”бҹ’бһҡбҹӮбһҘбһҸбһӮбһ·бһҸбһҗбҹ’бһӣбҹғбһ…бһ·бһ“бһ‘бҹ…бһҒбҹ’бһҳбҹӮбһҡ
        const url = `https://translate.googleapis.com/translate_a/single?client=gtx&sl=zh-CN&tl=km&dt=t&q=${encodeURIComponent(text)}`;
        const response = await fetch(url);
        const data = await response.json();
        
        let translatedText = "";
        data[0].forEach(item => { if(item[0]) translatedText += item[0]; });

        document.getElementById("translateResult").innerText = translatedText;
    } catch (error) {
        document.getElementById("translateResult").innerText = "вқҢ бһҳбһ·бһ“бһўбһ¶бһ…бһ”бһҖбһ”бҹ’бһҡбҹӮбһ”бһ¶бһ“бһ‘бҹҒ (бһҹбһјбһҳбһ–бһ·бһ“бһ·бһҸбҹ’бһҷбһўбҹҠбһёбһ“бһ’бһәбһҺбһ·бһҸбһҡбһ”бһҹбҹӢбһўбҹ’бһ“бһҖ)бҹ”";
        console.error(error);
    }
}
</script>
